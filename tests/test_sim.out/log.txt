dx := 1.5e-9
dy := 1.5e-9
dz := 1.5e-9
Nx := 25
Ny := 25
Nz := 1
setgridsize(Nx, Ny, Nz)
setcellsize(dx, dy, dz)
nd_size := 30e-9
nd := cylinder(nd_size, dz)
setgeom(nd)
m.SetInShape(nd, Neelskyrmion(1, -1))
Msat = 800e3
aex = 13e-12
alpha = 0.01
relax()
saveas(m, "stable")
t_sampl := 1e-10
tableadd(B_ext)
tableautosave(t_sampl)
autosave(m, t_sampl)
run(5 * t_sampl)
